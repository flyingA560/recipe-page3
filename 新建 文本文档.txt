    <script>
        const pages = {
            home: document.getElementById('page-home'),
            info: document.getElementById('page-info'),
            diet: document.getElementById('page-diet'),
            recipe: document.getElementById('page-recipe'),
            tips: document.getElementById('page-tips'),
            history: document.getElementById('page-history'),
            feedback: document.getElementById('page-feedback')
        };

        function showPage(pageName) {
            Object.keys(pages).forEach(key => pages[key].classList.add('hidden'));
            pages[pageName].classList.remove('hidden');
            document.getElementById('nav-home').classList.toggle('text-pink-300', pageName === 'home');
            document.getElementById('nav-home').classList.toggle('text-textLight', pageName !== 'home');
            document.getElementById('nav-history').classList.toggle('text-pink-300', pageName === 'history');
            document.getElementById('nav-history').classList.toggle('text-textLight', pageName !== 'history');
        }

        // ===================== BMI 计算功能 =====================
        function calculateBMI() {
            let height = parseFloat(document.getElementById('height').value) / 100;
            let weight = parseFloat(document.getElementById('weight').value);
            let bmi = weight / (height * height);
            bmi = bmi.toFixed(2);

            let level = '';
            let color = '';
            if(bmi < 18.5) {
                level = '偏瘦';
                color = 'text-blue-500';
            } else if(bmi < 24) {
                level = '正常';
                color = 'text-green-500';
            } else if(bmi < 28) {
                level = '超重';
                color = 'text-orange-500';
            } else {
                level = '肥胖';
                color = 'text-red-500';
            }

            document.getElementById('bmi-value').textContent = bmi;
            document.getElementById('bmi-level').textContent = 'BMI 结果：' + level;
            document.getElementById('bmi-level').className = 'text-[17px] font-medium ' + color;
        }

        // 输入时自动计算 BMI
        document.getElementById('height').addEventListener('input', calculateBMI);
        document.getElementById('weight').addEventListener('input', calculateBMI);

        const recipes = {
            breakfast: [{name:"营养均衡早餐",ingredients:"全麦面包2片、鸡蛋1个、牛奶200ml、苹果1个",calories:"320大卡",protein:"15g",carbs:"45g",vitamins:"丰富"},{name:"活力早餐",ingredients:"燕麦粥、蓝莓、酸奶、核桃",calories:"300大卡",protein:"12g",carbs:"48g",vitamins:"丰富"}],
            lunch: [{name:"减脂午餐",ingredients:"糙米饭、鸡胸肉、西兰花、番茄",calories:"450大卡",protein:"35g",carbs:"40g",vitamins:"适量"},{name:"营养午餐",ingredients:"杂粮饭、清蒸鱼、胡萝卜、青菜",calories:"420大卡",protein:"38g",carbs:"35g",vitamins:"丰富"}],
            dinner: [{name:"轻食晚餐",ingredients:"红薯、豆腐、菠菜、鸡蛋",calories:"350大卡",protein:"25g",carbs:"30g",vitamins:"丰富"},{name:"健康晚餐",ingredients:"玉米、虾仁、黄瓜、芹菜",calories:"320大卡",protein:"28g",carbs:"25g",vitamins:"适量"}]
        };

        document.querySelectorAll('.btn-back').forEach(btn => {
            btn.addEventListener('click', () => showPage(btn.dataset.back));
        });

        document.getElementById('btn-generate').addEventListener('click', () => {
            calculateBMI();

        });
        document.getElementById('btn-next').addEventListener('click', () => showPage('diet'));
        document.getElementById('btn-tips').addEventListener('click', () => showPage('tips'));
        document.getElementById('btn-back-to-recipe').addEventListener('click', () => showPage('recipe'));
        document.getElementById('btn-feedback').addEventListener('click', () => showPage('feedback'));
        document.getElementById('nav-home').addEventListener('click', () => showPage('home'));
        document.getElementById('nav-history').addEventListener('click', () => showPage('history'));

        function generateRecipe() {
            const meal = document.querySelector('input[name="meal"]:checked').value;
            const r = recipes[meal][Math.floor(Math.random()*recipes[meal].length)];
            document.getElementById('recipe-name').textContent = r.name;
            document.getElementById('recipe-ingredients').textContent = r.ingredients;
            document.getElementById('recipe-calories').textContent = r.calories;
            document.querySelector('#recipe-card .flex').innerHTML = `
                <span class="bg-pink-50 px-4 py-1 rounded-full shadow-sm">蛋白质：${r.protein}</span>
                <span class="bg-blue-50 px-4 py-1 rounded-full shadow-sm">碳水：${r.carbs}</span>
                <span class="bg-green-50 px-4 py-1 rounded-full shadow-sm">维生素：${r.vitamins}</span>`;
            showPage('recipe');
        }

        document.getElementById('btn-generate-recipe').addEventListener('click', generateRecipe);
        document.getElementById('btn-regenerate').addEventListener('click', generateRecipe);

        document.querySelectorAll('.fa-star').forEach(star => {
            star.addEventListener('click', () => {
                const v = star.dataset.rating;
                document.querySelectorAll('.fa-star').forEach(s => {
                    s.classList.toggle('text-yellow-400', s.dataset.rating <= v);
                    s.classList.toggle('text-gray-300', s.dataset.rating > v);
                });
            });
        });

        document.getElementById('btn-submit-feedback').addEventListener('click', () => {
            document.getElementById('feedback-success').classList.remove('hidden');
        });
        document.getElementById('btn-close-feedback').addEventListener('click', () => {
            document.getElementById('feedback-success').classList.add('hidden');
            showPage('feedback');
        });

        calculateBMI();
        showPage('home');
